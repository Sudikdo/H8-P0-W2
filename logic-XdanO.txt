function xo(str) {
  var x=0
  var o=0
  for (var j=0; j < str.length; j++){
    if(str[j]==='o'){
      o++
    }else if(str[j]==='x'){
      x++
    }
  }
}
if (o===x){
  return true
}else {
  return false
}

// TEST CASES
console.log(xo('xoxoxo')); // true
console.log(xo('oxooxo')); // false
console.log(xo('oxo')); // false
console.log(xo('xxxooo')); // true
console.log(xo('xoxooxxo')); // true
